LFS Notes

Initial Steps:

  • USB Drive > (4 GB)
  • Fedora Media Writer
  • Dell PC: (F2 - Setup | F12 - Boot Options)
  • Boot Mode: UEFI
  • Disable - Secure Boot
  • Grub -> Boot Fedora Workstation -> Try Fedora
  • Open Terminal
  • $ pwd -> /home/liveuser
  • $ touch version-check.sh
  • $ gnome-text-editor version-check.sh
                
Paste the Following:

                
                    #!/bin/bash
                    # Simple script to list version numbers of critical development tools
                    export LC_ALL=C
                    bash --version | head -n1 | cut -d" " -f2-4
                    MYSH=$(readlink -f /bin/sh)
                    echo "/bin/sh -> $MYSH"
                    echo $MYSH | grep -q bash || echo "ERROR: /bin/sh does not point to bash"
                    unset MYSH
                    
                    echo -n "Binutils: "; ld --version | head -n1 | cut -d" " -f3-
                    bison --version | head -n1
                    
                    if [ -h /usr/bin/yacc ]; then
                    echo "/usr/bin/yacc -> `readlink -f /usr/bin/yacc`";
                    elif [ -x /usr/bin/yacc ]; then
                    echo yacc is `/usr/bin/yacc --version | head -n1`
                    else
                    echo "yacc not found"
                    fi
                    
                    echo -n "Coreutils: "; chown --version | head -n1 | cut -d")" -f2
                    diff --version | head -n1
                    find --version | head -n1
                    gawk --version | head -n1
                    
                    if [ -h /usr/bin/awk ]; then
                    echo "/usr/bin/awk -> `readlink -f /usr/bin/awk`";
                    elif [ -x /usr/bin/awk ]; then
                    echo awk is `/usr/bin/awk --version | head -n1`
                    else
                    echo "awk not found"
                    fi
                    
                    gcc --version | head -n1
                    g++ --version | head -n1
                    grep --version | head -n1
                    gzip --version | head -n1
                    cat /proc/version
                    m4 --version | head -n1
                    make --version | head -n1
                    patch --version | head -n1
                    echo Perl `perl -V:version`
                    python3 --version
                    sed --version | head -n1
                    tar --version | head -n1
                    makeinfo --version | head -n1 # texinfo version
                    xz --version | head -n1
                    
                    echo 'int main(){}' > dummy.c && g++ -o dummy dummy.c
                    if [ -x dummy ]
                    then echo "g++ compilation OK";
                    else echo "g++ compilation failed"; fi
                    rm -f dummy.c dummy
                
                
$ bash version-check.sh

Verify UEFI Enabled

                
                    $ ls /sys/firmware/efi/efivars
                    If directory exists, system is UEFI enabled.
                
                
UEFI Enabled Partition Set-Up

                
                    $ fdisk -l
                
                
                    $ fdisk /dev/sda (targeted disk)
                    Command 'd' - To delete partition
                    Command 'n' - To create new partition
                
                
EFI (ESP) Partition for UEFI

                
                    : n 
                    : p (default)
                    : 1 (default)
                    : 2048 (default)
                    : +512M 
                    : t
                    : L
                    : ef (or EFI option)
                
                
Root Partition

                
                    : n
                    : p (default)
                    : 2 (default)
                    : 1050624(default)
                    : 52428799 (default)
                
                
Write Changes

                
                    : w
                
                
Create filesystem for UEFI

                
                    $ mkfs.fat -F32 /dev/sda1
                    $ mkfs.ext4 /dev/sda2
                
                
Setting $LFS Variable

                
                    $ export LFS=/mnt/lfs
                    $ echo $LFS
                
                
Mount New Partition

                
                    $ mkdir -pv $LFS
                    $ mount -v -t ext4 /dev/sda2 $LFS
